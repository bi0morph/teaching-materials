9 < 5;		// false

12 != 4;	// true

рекомендуется использовать всегда "==="

Сложные условия


true && false;		// false
true && true;		// true
false && false;		// false
false || true;		// true
false || false;		// false
true || true;			// true

&& - бинарное "И", возвращает true, если оба значение true

|| - бинарное "ИЛИ", возвращает true, если хотя бы значение true

Число 0, пустая строка "", null и undefined, а также NaN являются false,


var z, n;

n = 'number' * 2;

var x = 0 || '' || null || z || n || 2015;

console.log(x);
